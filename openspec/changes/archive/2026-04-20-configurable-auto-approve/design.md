## Implementation Steps

### Step 1: Settings.swift — Add configuration infrastructure

**File**: `Sources/CodeIsland/Settings.swift`

1. Add to `SettingsKey`:
   ```swift
   static func autoApproveTool(_ tool: String) -> String { "autoApproveTool_\(tool)" }
   ```

2. Add to `SettingsDefaults`:
   ```swift
   static let autoApproveDefaultTools: Set<String> = [
       "TaskCreate", "TaskUpdate", "TaskGet", "TaskList", "TaskOutput", "TaskStop",
       "TodoRead", "TodoWrite", "EnterPlanMode", "ExitPlanMode",
   ]
   ```

3. Add to `SettingsManager`:
   ```swift
   static let allAutoApproveTools = Array(autoApproveDefaultTools).sorted()

   func isAutoApproveTool(_ tool: String) -> Bool {
       let key = SettingsKey.autoApproveTool(tool)
       if defaults.object(forKey: key) == nil {
           return SettingsDefaults.autoApproveDefaultTools.contains(tool)
       }
       return defaults.bool(forKey: key)
   }

   func setAutoApproveTool(_ tool: String, enabled: Bool) {
       defaults.set(enabled, forKey: SettingsKey.autoApproveTool(tool))
   }
   ```

### Step 2: SettingsView.swift — Add toggle UI in BehaviorPage

**File**: `Sources/CodeIsland/SettingsView.swift`

Add new Section after the `sessions` Section in `BehaviorPage`:
```swift
Section(l10n["auto_approve_tools"]) {
    ForEach(SettingsManager.allAutoApproveTools, id: \.self) { tool in
        Toggle(tool, isOn: Binding(
            get: { SettingsManager.shared.isAutoApproveTool(tool) },
            set: { SettingsManager.shared.setAutoApproveTool(tool, enabled: $0) }
        ))
    }
}
```

### Step 3: HookServer.swift — Replace hardcoded list with dynamic check

**File**: `Sources/CodeIsland/HookServer.swift`

1. Remove the `autoApproveTools` static constant (lines 118-123)
2. Change the check at line 153:
   ```swift
   // Before:
   if let toolName = event.toolName, Self.autoApproveTools.contains(toolName) {
   // After:
   if let toolName = event.toolName, SettingsManager.shared.isAutoApproveTool(toolName) {
   ```
